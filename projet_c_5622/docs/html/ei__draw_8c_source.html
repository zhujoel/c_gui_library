683 ei_linked_point_t* arc(const ei_point_t centre, float rayon, int angle_debut, int angle_fin){
684  // coordonnées x et y en fonction des paramètres
685  int x = 0;
686  int y = 0;
687  int nbElems = abs(angle_debut-angle_fin);
688  ei_linked_point_t* pts = malloc(sizeof(ei_linked_point_t)*nbElems+1);
689  int i = 0;
690  // calcule l'orientation de l'arc
691  int incrementi = angle_debut < angle_fin ? 1 : -1;
692  int index_current = 0;
693  // on dessine un point pour chaque angle
694  while(index_current < (nbElems-2)){
695  x = rayon * cos((angle_debut+i)*(PI/180)) + centre.x;
696  y = rayon * sin((angle_debut+i)*(PI/180)) + centre.y;
697  pts[index_current].point.x = x; pts[index_current].point.y = y; pts[index_current].next = &pts[index_current+1];
698  i += incrementi;
699  index_current++;
700  }
701 
702  // avant dernier point pointe vers dernier
703  pts[index_current-2].next = &pts[index_current-1];
704  // dernier point
705  x = rayon * cos((i+angle_debut)*(PI/180)) + centre.x;
706  y = rayon * sin((i+angle_debut)*(PI/180)) + centre.y;
707  pts[index_current-1].point.x = x; pts[index_current-1].point.y = y; pts[index_current-1].next = NULL;
708 
709  return pts;
