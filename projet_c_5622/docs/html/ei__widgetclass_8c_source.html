554  && event->param.mouse.button_number == 1 )
555  {
556  ei_event_set_active_widget(NULL);
557  return EI_TRUE;
558  }
559  else if (event->type == ei_ev_mouse_move)
560  {
561  int deltax = event->param.mouse.where.x - position_precedente.x;
562  int deltay = event->param.mouse.where.y - position_precedente.y;
563  if (action == ei_deplace)
564  {
565 
566  widget->placer_params->x_data += deltax;
567  widget->placer_params->y_data += deltay;
568  ei_placer_run(widget);
569  }
570  else if (action == ei_redim)
571  {
572  if (widgettoplevel->resizable == ei_axis_x || widgettoplevel->resizable == ei_axis_both)
573  {
574  widget->placer_params->w_data += deltax;
575  }
576  if (widgettoplevel->resizable == ei_axis_y || widgettoplevel->resizable == ei_axis_both)
577  {
578  widget->placer_params->h_data += deltay;
579  }
580  ei_placer_run(widget);
581  }
582 
583  if(widget->placer_params->y_data < 0){
584  widget->placer_params->y_data = 0;
585  }else if(widget->placer_params->y_data + widget->placer_params->h_data > ei_app_root_widget()->screen_location.size.height){
586  widget->placer_params->y_data = ei_app_root_widget()->screen_location.size.height - widget->placer_params->h_data;
587  }
588  if(widget->placer_params->x_data < 0){
589  widget->placer_params->x_data = 0;
590  }else if(widget->placer_params->x_data + widget->placer_params->w_data > ei_app_root_widget()->screen_location.size.width){
591  widget->placer_params->x_data = ei_app_root_widget()->screen_location.size.width - widget->placer_params->w_data;
592  }
593  position_precedente = event->param.mouse.where;
594  return EI_TRUE;
595  }
596  }
597  return EI_FALSE;
598 }
